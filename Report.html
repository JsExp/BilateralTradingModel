clear;
% num: Number of agents
% beta: Client stay probability
% k: Server's fixed bid
% s: Surplus
% c_serve: Cost of server if trade success
% c_lose: Cost of client if trade failed
INIT = struct('num', 1000, 'beta', 0.98, ...
'k', 7, 's', 8, 'alpha', 0.5, ...
'c_serve', 6, 'c_lose', 0.5);
Graph = imread('MFG.png');
figure;
imshow(Graph);
ValueFunction = imread('ValueFunction.png');
figure;
imshow(ValueFunction);
title('Value Function')
% Reinforcemrnt Learning --> For value function
% state: state
% / \
% action(1): client* *server --> [p_client==0.5, p_server==0.5]
% action(2): (client policy) <-- / \ z/ \(1-z) --> (z is related to policy)
% next_state: not bid fail suc
clf;
state_max = 30; % For finite interation
line = 1;
precent = [];
for alpha = 0.1:0.2:0.5
INIT.alpha = alpha;
rwd = [-INIT.c_lose; INIT.s-INIT.k; 0; INIT.k-INIT.c_serve]; % Reward of [not, bid, fail, suc]
delta_bgt = transpose(rwd); % Budget Delta
delta_bgt(1) = 0;
Binit = 7;
state = 0:Binit; % Posiable budget initiation
value = zeros(1,Binit+1); % Store all state's value
z = [0, 1];
error = 1;
interation =0;
% for i=1:1000
while error>0.0001 || interation < 100
% State(budget) interation(update) number
st_new = []; % New state accured during bid process
z(2)=z(1);
z(1)=Prob_not(INIT, state);
for st=state % Update all state have beent know
if st < INIT.k-INIT.s/(1+INIT.alpha)
policy = 0.5*[1 0 z(1) 1-z(1)]; % [not, bid]
elseif st < INIT.k-(INIT.s-INIT.k)/(INIT.alpha)
if INIT.s-INIT.k-INIT.alpha*(INIT.k-st)<0
policy = 0.5*[1 0 z(1) 1-z(1)]; % [not, bid]
end
else
policy = 0.5*[0 1 z(1) 1-z(1)]; % [not, bid]
end
st_next = roundn(st+policy.*delta_bgt, -1);
value_next = [];
for stn=st_next
% tmp = 1;
if stn > state_max
% st_next(tmp)=state_max;
value_next = [value_next, value(state_max)];
else
if ~ismember(stn, state)
state=[state, stn];
value = [value, 0];
st_new = [st_new, stn];
end
value_next = [value_next, value(state==stn)];
end
% tmp = tmp + 1;
end
value(state==st) = policy*(rwd+0.98*transpose(value_next));
end
s_state = sort(state);
s_value = [];
len = length(value);
for si = 1:len
s_value = [s_value, value(state==s_state(si))];
end
error = s_value(len-round(len/10):len)*ones(round(len/10)+1,1)...
- s_value(len-round(len/10)-1:len-1)*ones(round(len/10)+1,1);
error = abs(error);
interation = interation +1;
end
% value_exp = [value_exp, s_value(len)]
% end
if line ==1
plot(s_state, s_value, '-ro');
elseif line ==2
plot(s_state, s_value, '-b');
else
plot(s_state, s_value, '-g');
end
line = line+1;
hold on;
precent = [precent, s_value(len)/s_value(1)];
% plot(6:8, value_exp, '-o');
end
legend('alpha:0.1','alpha:0.3','alpha:0.5');
legend('alpha:0.1','alpha:0.3','alpha:0.5');
title('Value Interation, k=7')
xlabel('Budget(state)')
ylabel('value')
figure;
plot([0.1 0.3 0.5], precent, '-*');
title('High budget/Low budget');
xlabel('alpha')
ylabel('percent')
figure;
clf;
fig = 1;
for price = [6.5 7 7.1]
INIT.k = price;
figure(fig);
clf;
fig = fig +1;
line = 1;
for alph = 1:2:5
INIT.alpha = alph/10;
budgets = BudgetGenerator([0, 5], INIT.num);
stay = StayStatusGen(INIT.beta, INIT.num);
rwd_s = [0; INIT.k-INIT.c_serve]; % Reward of [not, bid, fail, suc]
z_store = ones(1, 100);
for loop=1:100
budget_c = budgets(1:INIT.num/2);
budget_s = budgets(INIT.num/2+1:end);
z=Prob_not(INIT, budgets);
i=1;
for st=budget_c % Update all state have beent know
rwd_c = [0; INIT.s-INIT.k-INIT.alpha*(INIT.k-st)];
if st < INIT.k-INIT.s/(1+INIT.alpha)
policy = [1 0];
elseif st < INIT.k-(INIT.s-INIT.k)/(INIT.alpha)
if rwd_c(2)<0
policy = [1 0];
end
else
policy = [0 1];
end
st;
policy;
budget_c(i)=budget_c(i) + policy*rwd_c;
budget_s(i)=budget_s(i) + policy*rwd_s;
i=i+1;
end
budgets = [budget_c, budget_s];
shuffle = randperm(INIT.num);
budgets = budgets(shuffle);
z_store(loop)=z;
end
if line == 1
plot(1:100,z_store, '-ro');
elseif line == 2
plot(1:100,z_store, '-b*');
elseif line == 3
plot(1:100,z_store, '-g>');
end
hold on;
line = line +1;
end
legend('alpha:0.1','alpha:0.3','alpha:0.5');
title(['k=',num2str(price)]);
xlabel('Interation');
ylabel('Failed probability');
end
function z = Prob_not(INIT,state)
count = 0;
for st=state
rwd_c = [0; INIT.s-INIT.k-INIT.alpha*(INIT.k-st)];
if st < INIT.k-INIT.s/(1+INIT.alpha)
count = count+1;
elseif st < INIT.k-(INIT.s-INIT.k)/(INIT.alpha)
if rwd_c(2)<0
count = count+1;
end
end
end
z = count/length(state);
end
function b = BudgetGenerator(scop, len)
b = randi(scop, 1, len);
end
function s = StayStatusGen(prob, len)
% 0 -> stay; 1 -> leave
s = randsrc(1, len, [0, 1; prob, 1-prob]);
end
